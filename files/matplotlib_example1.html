Matplotlib: US Cumulative GDP Growth, Population Growth (2013-2017)¶

Source data: State GDP: BEA (chained 2009 dollars), MSA Population (Census, Apr 2010 - July 2017)

Goal: to compare metropolitan cumulative GDP growth from 2013 to 2017 and compare Ann Arbor's growth against the baseline (US) versus similarly sized metropolitan areas based on population at the end of 2017. The first chart was specifically a quick Coursera final project in learning to use matplotlib, but I ended up being curious as to what else I could put together with this data outside of just Ann Arbor.

This was also an exercise to see if I could get my speed up faster than just doing this in Excel. It's not, but you really can't beat the fact that this is basically putting together a research report in Word plus making Excel charts at the same time. And that's pretty awesome!

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# importing the population and gdp data
metro_pop = pd.read_csv("https://raw.githubusercontent.com/stuffofminsun/CouseraAppliedPlotting/master"
                        "/cbsa-est2017-alldata.csv", encoding='cp1252')
metro_gdp = pd.read_csv("https://raw.githubusercontent.com/stuffofminsun/CouseraAppliedPlotting/master"
                        "/metropolitanareaschainedgdp.csv", skiprows=4, delimiter=',')

# cleaning up GDP
metro_gdp.drop(metro_gdp.index[-4:], inplace=True)
metro_gdp.drop(['GeoFips', '2001', '2002', '2003', '2004', '2005', '2006', '2007', '2008', '2009', 
                '2010', '2011', '2012', '2014', '2015', '2016'], axis = 1, inplace=True)
metro_gdp['5Y Change'] = (metro_gdp['2017']/metro_gdp['2013'])-1
metro_gdp['GeoName'] = metro_gdp['GeoName'].str.replace(r" \(.*\)","")
metro_gdp.set_index('GeoName', inplace=True)

# cleaning up population and getting average population per area based on 2010-2017 estimates
column_names = ['POPESTIMATE2013', 'POPESTIMATE2014', 'POPESTIMATE2015', 'POPESTIMATE2016', 'POPESTIMATE2017']
metro_pop_all = metro_pop[metro_pop['LSAD'] == 'Metropolitan Statistical Area']
metro_pop['AVGPOPESTIMATE'] = metro_pop_all[column_names].mean(axis=1)
metro_pop = metro_pop[['NAME', 'AVGPOPESTIMATE']]
metro_pop.set_index('NAME', inplace=True)

combined = pd.merge(metro_gdp, metro_pop, how='inner', left_index=True, right_index=True)
blah  = pd.merge(metro_gdp, metro_pop, how='left', left_index=True, right_index=True)
annarbor_upperrange = combined[combined.index == 'Ann Arbor, MI']['AVGPOPESTIMATE'].values[0] + 30000
annarbor_lowerrange = combined[combined.index == 'Ann Arbor, MI']['AVGPOPESTIMATE'].values[0] - 30000
combined = combined[combined['AVGPOPESTIMATE'] < annarbor_upperrange]
combined = combined[combined['AVGPOPESTIMATE'] > annarbor_lowerrange]
combined.sort_values('AVGPOPESTIMATE', inplace=True)

# calculating peer group growth
group_2013 = combined['2013'].sum()
group_2017 = combined['2017'].sum()
group_5Y = (group_2017/group_2013)-1

# charting 
sns.set()
fig, ax = plt.subplots(figsize=(12,7))

plt.title('Cumulative 5Y GDP Growth (2013-2017)\n Ann Arbor GDP vs. Similarly Sized Metropolitan Statistical '
          'Areas\n (Population +/- 30,000) ')
plt.ylabel('Average Population (2013-2017)')
plt.xlabel('GDP Growth, 2013-2017')
plt.plot(combined['5Y Change'], combined['AVGPOPESTIMATE'], 'o', label='Peer MSAs')
plt.plot(combined[combined.index == 'Ann Arbor, MI']['5Y Change'].values[0], 
         combined[combined.index == 'Ann Arbor, MI']['AVGPOPESTIMATE'].values[0], 'o', color='red')

plt.annotate('Ann Arbor, MI', (combined[combined.index == 'Ann Arbor, MI']['5Y Change'].values[0]-.04, 
         combined[combined.index == 'Ann Arbor, MI']['AVGPOPESTIMATE'].values[0]-3500))

# set 5Y GDP growth for United States & peer group 
plt.axvspan(0.095968, 0.095968, color='black', label='All \nUS 2013-2017 \n Growth')
plt.axvspan(group_5Y, group_5Y, color='blue', label='Peer Group \n 2013-2017 \n Growth')
plt.legend(loc='best')
plt.show()

all_combined = pd.merge(metro_gdp, metro_pop, how='inner', left_index=True, right_index=True)
all_combined['Quintile'] = pd.qcut(all_combined['AVGPOPESTIMATE'], 5, labels=False)
quintiles = all_combined.groupby('Quintile').agg({'2013': 'sum', '2017': 'sum', '5Y Change': 'mean', 
                                                  'AVGPOPESTIMATE': ['mean', np.min, np.max]})
quintiles.columns = ['2013', '2017', '5Y Change', 'AVGPOPESTIMATE', 'Minimum Pop', 'Maximum Pop']
quintiles['5Y Change'] = quintiles['2017']/quintiles['2013']-1

fig, ax = plt.subplots(figsize=(12,7))
plt.title('Cumulative 5Y GDP Growth (2013-2017) by Quintile')
plt.ylabel('Average Population (2013-2017)')
plt.xlabel('Cumulative GDP Growth, 2013-2017')

plt.plot(quintiles['5Y Change'], quintiles['AVGPOPESTIMATE'], 'o', label='Quintile')
for i in range(5):
    plt.annotate("Pop Range: {:,}-{:,}".format(quintiles[quintiles.index==i]['Minimum Pop'].values[0], 
                                     quintiles[quintiles.index==i]['Maximum Pop'].values[0]), 
                 (quintiles[quintiles.index==i]['5Y Change'].values[0], 
                  quintiles[quintiles.index==i]['AVGPOPESTIMATE'].values[0]))

plt.axvspan(0.095968, 0.095968, color='black', label='All \nUS 2013-2017 \n Growth')
plt.legend(loc='best')
plt.show()
